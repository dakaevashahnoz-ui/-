<script>
  const proverbs = [
    { start: "Еңбек етсең ерінбей...", end: "тояды қарның тіленбей" },
    { start: "Оқу — білім бұлағы...", end: "білім — өмір шырағы" },
    { start: "Досы көпті жау алмайды...", end: "ақылы көпті дау алмайды" },
    { start: "Тіл тас жарады...", end: "тас жармаса бас жарады" }
  ];

  let current;

  function loadProverb() {
    current = proverbs[Math.floor(Math.random() * proverbs.length)];
    document.getElementById("proverb").innerText = current.start;
    document.getElementById("answer").value = "";
    document.getElementById("result").innerText = "";
  }

  function checkAnswer() {
    let userAnswer = document.getElementById("answer").value.toLowerCase().trim();
    let correct = current.end.toLowerCase();

    if (userAnswer === correct) {
      document.getElementById("result").innerText = "✅ Дұрыс!";
      document.getElementById("result").style.color = "green";
    } else {
      document.getElementById("result").innerText = "❌ Қате! Дұрыс жауап: " + current.end;
      document.getElementById("result").style.color = "red";
    }
  }

  function nextProverb() {
    loadProverb();
  }

  loadProverb();
</script>
